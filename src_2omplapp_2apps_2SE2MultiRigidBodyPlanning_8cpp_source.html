15 ompl::app::SE2MultiRigidBodyPlanning::SE2MultiRigidBodyPlanning(unsigned int n) :

41 ompl::base::ScopedState<> ompl::app::SE2MultiRigidBodyPlanning::getDefaultStartState(void) const
42 {
43  base::ScopedState<> st(getStateSpace());
44  base::CompoundStateSpace::StateType* c_st = st.get()->as<base::CompoundStateSpace::StateType>();
45  for (unsigned int i = 0; i < n_; ++i)
46  {
47  aiVector3D s = getRobotCenter(i);
48  base::SE2StateSpace::StateType* sub = c_st->as<base::SE2StateSpace::StateType>(i);
49  sub->setX(s.x);
50  sub->setY(s.y);
51  sub->setYaw(0.0);
52  }
53  return st;
54 }
