54 void ompl::app::KinematicCarPlanning::ode(const control::ODESolver::StateType& q, const control::Control *ctrl, control::ODESolver::StateType& qdot)
55 {
56  const double *u = ctrl->as<control::RealVectorControlSpace::ControlType>()->values;
57 
58  // zero out qdot
59  qdot.resize (q.size (), 0);
60 
61  qdot[0] = u[0] * cos(q[2]);
62  qdot[1] = u[0] * sin(q[2]);
63  qdot[2] = u[0] * lengthInv_ * tan(u[1]);
64 }
