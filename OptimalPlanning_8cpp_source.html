91 // Our "collision checker". For this demo, our robot's state space
92 // lies in [0,1]x[0,1], with a circular obstacle of radius 0.25
93 // centered at (0.5,0.5). Any states lying in this circular region are
94 // considered "in collision".
95 class ValidityChecker : public ob::StateValidityChecker
96 {
97 public:
98  ValidityChecker(const ob::SpaceInformationPtr& si) :
99  ob::StateValidityChecker(si) {}
100 
101  // Returns whether the given state's position overlaps the
102  // circular obstacle
103  bool isValid(const ob::State* state) const
104  {
105  return this->clearance(state) > 0.0;
106  }
107 
108  // Returns the distance from the given state's position to the
109  // boundary of the circular obstacle.
110  double clearance(const ob::State* state) const
111  {
112  // We know we're working with a RealVectorStateSpace in this
113  // example, so we downcast state into the specific type.
114  const ob::RealVectorStateSpace::StateType* state2D =
115  state->as<ob::RealVectorStateSpace::StateType>();
116 
117  // Extract the robot's (x,y) position from its state
118  double x = state2D->values[0];
119  double y = state2D->values[1];
120 
121  // Distance formula between two points, offset by the circle's
122  // radius
123  return sqrt((x-0.5)*(x-0.5) + (y-0.5)*(y-0.5)) - 0.25;
124  }
125 };

205 void plan(double runTime, optimalPlanner plannerType, planningObjective objectiveType, std::string outputFile)
206 {
207  // Construct the robot state space in which we're planning. We're
208  // planning in [0,1]x[0,1], a subset of R^2.
209  ob::StateSpacePtr space(new ob::RealVectorStateSpace(2));
210 
211  // Set the bounds of space to be in [0,1].
212  space->as<ob::RealVectorStateSpace>()->setBounds(0.0, 1.0);
213 
214  // Construct a space information instance for this state space
215  ob::SpaceInformationPtr si(new ob::SpaceInformation(space));
216 
217  // Set the object used to check which states in the space are valid
218  si->setStateValidityChecker(ob::StateValidityCheckerPtr(new ValidityChecker(si)));
219 
220  si->setup();
221 
222  // Set our robot's starting state to be the bottom-left corner of
223  // the environment, or (0,0).
224  ob::ScopedState<> start(space);
225  start->as<ob::RealVectorStateSpace::StateType>()->values[0] = 0.0;
226  start->as<ob::RealVectorStateSpace::StateType>()->values[1] = 0.0;
227 
228  // Set our robot's goal state to be the top-right corner of the
229  // environment, or (1,1).
230  ob::ScopedState<> goal(space);
231  goal->as<ob::RealVectorStateSpace::StateType>()->values[0] = 1.0;
232  goal->as<ob::RealVectorStateSpace::StateType>()->values[1] = 1.0;
233 
234  // Create a problem instance
235  ob::ProblemDefinitionPtr pdef(new ob::ProblemDefinition(si));
236 
237  // Set the start and goal states
238  pdef->setStartAndGoalStates(start, goal);
239 
240  // Create the optimization objective specified by our command-line argument.
241  // This helper function is simply a switch statement.
242  pdef->setOptimizationObjective(allocateObjective(si, objectiveType));
243 
244  // Construct the optimal planner specified by our command line argument.
245  // This helper function is simply a switch statement.
246  ob::PlannerPtr optimizingPlanner = allocatePlanner(si, plannerType);
247 
248  // Set the problem instance for our planner to solve
249  optimizingPlanner->setProblemDefinition(pdef);
250  optimizingPlanner->setup();
251 
252  // attempt to solve the planning problem in the given runtime
253  ob::PlannerStatus solved = optimizingPlanner->solve(runTime);
254 
255  if (solved)
256  {
257  // Output the length of the path found
258  std::cout
259  << optimizingPlanner->getName()
260  << " found a solution of length "
261  << pdef->getSolutionPath()->length()
262  << " with an optimization objective value of "
263  << pdef->getSolutionPath()->cost(pdef->getOptimizationObjective()) << std::endl;
264 
265  // If a filename was specified, output the path as a matrix to
266  // that file for visualization
267  if (!outputFile.empty())
268  {
269  std::ofstream outFile(outputFile.c_str());
270  boost::static_pointer_cast<og::PathGeometric>(pdef->getSolutionPath())->
271  printAsMatrix(outFile);
272  outFile.close();
273  }
274  }
275  else
276  std::cout << "No solution found." << std::endl;
277 }

334 class ClearanceObjective : public ob::StateCostIntegralObjective
335 {
336 public:
337  ClearanceObjective(const ob::SpaceInformationPtr& si) :
338  ob::StateCostIntegralObjective(si, true)
339  {
340  }
341 
342  // Our requirement is to maximize path clearance from obstacles,
343  // but we want to represent the objective as a path cost
344  // minimization. Therefore, we set each state's cost to be the
345  // reciprocal of its clearance, so that as state clearance
346  // increases, the state cost decreases.
347  ob::Cost stateCost(const ob::State* s) const
348  {
349  return ob::Cost(1 / si_->getStateValidityChecker()->clearance(s));
350  }
351 };

406 bool argParse(int argc, char** argv, double* runTimePtr, optimalPlanner *plannerPtr, planningObjective *objectivePtr, std::string *outputFilePtr)
407 {
408  namespace bpo = boost::program_options;
409 
410  // Declare the supported options.
411  bpo::options_description desc("Allowed options");
412  desc.add_options()
413  ("help,h", "produce help message")
414  ("runtime,t", bpo::value<double>()->default_value(1.0), "(Optional) Specify the runtime in seconds. Defaults to 1 and must be greater than 0.")
415  ("planner,p", bpo::value<std::string>()->default_value("RRTstar"), "(Optional) Specify the optimal planner to use, defaults to RRTstar if not given. Valid options are BITstar, CForest, FMTstar, InformedRRTstar, PRMstar, and RRTstar.") //Alphabetical order
416  ("objective,o", bpo::value<std::string>()->default_value("PathLength"), "(Optional) Specify the optimization objective, defaults to PathLength if not given. Valid options are PathClearance, PathLength, ThresholdPathLength, and WeightedLengthAndClearanceCombo.") //Alphabetical order
417  ("file,f", bpo::value<std::string>()->default_value(""), "(Optional) Specify an output path for the found solution path.")
418  ("info,i", bpo::value<unsigned int>()->default_value(0u), "(Optional) Set the OMPL log level. 0 for WARN, 1 for INFO, 2 for DEBUG. Defaults to WARN.");
419  bpo::variables_map vm;
420  bpo::store(bpo::parse_command_line(argc, argv, desc), vm);
421  bpo::notify(vm);
422 
423  // Check if the help flag has been given:
424  if (vm.count("help"))
425  {
426  std::cout << desc << std::endl;
427  return false;
428  }
429 
430  // Set the log-level
431  unsigned int logLevel = vm["info"].as<unsigned int>();
432 
433  // Switch to setting the log level:
434  if (logLevel == 0u)
435  {
436  ompl::msg::setLogLevel(ompl::msg::LOG_WARN);
437  }
438  else if (logLevel == 1u)
439  {
440  ompl::msg::setLogLevel(ompl::msg::LOG_INFO);
441  }
442  else if (logLevel == 2u)
443  {
444  ompl::msg::setLogLevel(ompl::msg::LOG_DEBUG);
445  }
446  else
447  {
448  std::cout << "Invalid log-level integer." << std::endl << std::endl << desc << std::endl;
449  return false;
450  }
451 
452  // Get the runtime as a double
453  *runTimePtr = vm["runtime"].as<double>();
454 
455  // Sanity check
456  if (*runTimePtr <= 0.0)
457  {
458  std::cout << "Invalid runtime." << std::endl << std::endl << desc << std::endl;
459  return false;
460  }
461 
462  // Get the specified planner as a string
463  std::string plannerStr = vm["planner"].as<std::string>();
464 
465  // Map the string to the enum
466  if (boost::iequals("BITstar", plannerStr))
467  {
468  *plannerPtr = PLANNER_BITSTAR;
469  }
470  else if (boost::iequals("CForest", plannerStr))
471  {
472  *plannerPtr = PLANNER_CFOREST;
473  }
474  else if (boost::iequals("FMTstar", plannerStr))
475  {
476  *plannerPtr = PLANNER_FMTSTAR;
477  }
478  else if (boost::iequals("InformedRRTstar", plannerStr))
479  {
480  *plannerPtr = PLANNER_INF_RRTSTAR;
481  }
482  else if (boost::iequals("PRMstar", plannerStr))
483  {
484  *plannerPtr = PLANNER_PRMSTAR;
485  }
486  else if (boost::iequals("RRTstar", plannerStr))
487  {
488  *plannerPtr = PLANNER_RRTSTAR;
489  }
490  else
491  {
492  std::cout << "Invalid planner string." << std::endl << std::endl << desc << std::endl;
493  return false;
494  }
495 
496  // Get the specified objective as a string
497  std::string objectiveStr = vm["objective"].as<std::string>();
498
