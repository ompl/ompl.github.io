61  double HaltonSequence1D::sample()
62  {
63  double f = 1, r = 0;
64  unsigned int i = i_;
65 
66  while (i > 0)
67  {
68  f /= base_;
69  r += f * (i % base_);
70  i = std::floor(i / base_);
71  }
72 
73  ++i_;
74  return r;
75  }

83  HaltonSequence::HaltonSequence(unsigned int dimensions, std::vector<unsigned int> bases)
84  : DeterministicSequence(dimensions), halton_sequences_1d_(dimensions)
85  {
86  if (bases.size() != dimensions)
87  {
88  OMPL_WARN("Number of bases does not match dimensions. Using first n primes instead.");
89  }
90  else
91  {
92  int i = 0;
93  for (auto base : bases)
94  {
95  halton_sequences_1d_[i].setBase(base);
96  i++;
97  }
98  }
99  }

111  void HaltonSequence::setBasesToPrimes()
112  {
113  // set the base of the halton sequences to the first n prime numbers, where n is dimensions
114  unsigned int current = 2;
115  for (unsigned int i = 0; i < dimensions_; i++)
116  {
117  current = boost::math::prime(i);
118  halton_sequences_1d_[i].setBase(current);
119  }
