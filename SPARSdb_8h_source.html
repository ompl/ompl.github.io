114  struct InterfaceData
115  {
117  base::State *pointA_;
118  base::State *pointB_;

121  base::State *sigmaA_;
122  base::State *sigmaB_;
123 
125  double d_;
126 
128  InterfaceData() :
129  pointA_(NULL),
130  pointB_(NULL),
131  sigmaA_(NULL),
132  sigmaB_(NULL),
133  d_(std::numeric_limits<double>::infinity())
134  {
135  }
136 
138  void clear(const base::SpaceInformationPtr& si)
139  {
140  if (pointA_)
141  {
142  si->freeState(pointA_);
143  pointA_ = NULL;
144  }
145  if (pointB_)
146  {
147  si->freeState(pointB_);
148  pointB_ = NULL;
149  }
150  if (sigmaA_)
151  {
152  si->freeState(sigmaA_);
153  sigmaA_ = NULL;
154  }
155  if (sigmaB_)
156  {
157  si->freeState(sigmaB_);
158  sigmaB_ = NULL;
159  }
160  d_ = std::numeric_limits<double>::infinity();
161  }
162 
164  void setFirst(const base::State *p, const base::State *s, const base::SpaceInformationPtr& si)
165  {
166  if (pointA_)
167  si->copyState(pointA_, p);
168  else
169  pointA_ = si->cloneState(p);
170  if (sigmaA_)
171  si->copyState(sigmaA_, s);
172  else
173  sigmaA_ = si->cloneState(s);
174  if (pointB_)
175  d_ = si->distance(pointA_, pointB_);
176  }
177 
179  void setSecond(const base::State *p, const base::State *s, const base::SpaceInformationPtr& si)
180  {
181  if (pointB_)
182  si->copyState(pointB_, p);
183  else
184  pointB_ = si->cloneState(p);
185  if (sigmaB_)
186  si->copyState(sigmaB_, s);
187  else
188  sigmaB_ = si->cloneState(s);
189  if (pointA_)
190  d_ = si->distance(pointA_, pointB_);
191  }
192  };
