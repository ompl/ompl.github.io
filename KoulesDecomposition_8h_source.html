50 static ompl::base::RealVectorBounds bounds3(const ompl::base::RealVectorBounds &bounds)
51 {
52 ompl::base::RealVectorBounds b(3);
53 std::copy(bounds.low.begin(), bounds.low.begin() + 3, b.low.begin());
54 std::copy(bounds.high.begin(), bounds.high.begin() + 3, b.high.begin());
55 return b;
56 }
57 void project(const ompl::base::State *s, std::vector<double> &coord) const override
58 {
59 const double* pos = s->as<ompl::base::RealVectorStateSpace::StateType>()->values;
60 coord.resize(3);
61 coord[0] = pos[0];
62 coord[1] = pos[1];
63 coord[2] = pos[2];
64 }
65
66 void sampleFullState(const ompl::base::StateSamplerPtr &sampler,
67 const std::vector<double>& coord, ompl::base::State *s) const override
68 {
69 double* pos = s->as<ompl::base::RealVectorStateSpace::StateType>()->values;
70 sampler->sampleUniform(s);
71 pos[0] = coord[0];
72 pos[1] = coord[1];
73 pos[2] = coord[2];
74 }
75};
