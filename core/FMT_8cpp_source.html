81 void ompl::geometric::FMT::setup()
82 {
83  if (pdef_)
84  {
85  /* Setup the optimization objective. If no optimization objective was
86  specified, then default to optimizing path length as computed by the
87  distance() function in the state space */
88  if (pdef_->hasOptimizationObjective())
89  opt_ = pdef_->getOptimizationObjective();
90  else
91  {
92  OMPL_INFORM("%s: No optimization objective specified. Defaulting to optimizing path length.", getName().c_str());
93  opt_.reset(new base::PathLengthOptimizationObjective(si_));
94  }
95  Open_.getComparisonOperator().opt_ = opt_.get();
96  Open_.getComparisonOperator().heuristics_ = heuristics_;
97 
98  if (!nn_)
99  nn_.reset(tools::SelfConfig::getDefaultNearestNeighbors<Motion*>(this));
100  nn_->setDistanceFunction(boost::bind(&FMT::distanceFunction, this, _1, _2));
101 
102  if (nearestK_ && !nn_->reportsSortedResults())
103  {
104  OMPL_WARN("%s: NearestNeighbors datastructure does not return sorted solutions. Nearest K strategy disabled.", getName().c_str());
105  nearestK_ = false;
106  }
107  }
108  else
109  {
110  OMPL_INFORM("%s: problem definition is not set, deferring setup completion...", getName().c_str());
111  setup_ = false;
112  }
113 }

164 void ompl::geometric::FMT::saveNeighborhood(Motion *m)
165 {
166  // Check to see if neighborhood has not been saved yet
167  if (neighborhoods_.find(m) == neighborhoods_.end())
168  {
169  std::vector<Motion*> nbh;
170  if (nearestK_)
171  nn_->nearestK(m, NNk_, nbh);
172  else
173  nn_->nearestR(m, NNr_, nbh);
174  if (!nbh.empty())
175  {
176  // Save the neighborhood but skip the first element, since it will be motion m
177  neighborhoods_[m] = std::vector<Motion*>(nbh.size() - 1, 0);
178  std::copy(nbh.begin() + 1, nbh.end(), neighborhoods_[m].begin());
179  }
180  else
181  {
182  // Save an empty neighborhood
183  neighborhoods_[m] = std::vector<Motion*>(0);
184  }
185  } // If neighborhood hadn't been saved yet
186 }

188 // Calculate the unit ball volume for a given dimension
189 double ompl::geometric::FMT::calculateUnitBallVolume(const unsigned int dimension) const
190 {
191  if (dimension == 0)
192  return 1.0;
193  else if (dimension == 1)
194  return 2.0;
195  return 2.0 * boost::math::constants::pi<double>() / dimension
196  * calculateUnitBallVolume(dimension - 2);
197 }

199 double ompl::geometric::FMT::calculateRadius(const unsigned int dimension, const unsigned int n) const
200 {
201  double a = 1.0 / (double)dimension;
202  double unitBallVolume = calculateUnitBallVolume(dimension);
203 
204  return radiusMultiplier_ * 2.0 * std::pow(a, a) * std::pow(freeSpaceVolume_ / unitBallVolume, a) * std::pow(log((double)n) / (double)n, a);
205 }

207 void ompl::geometric::FMT::sampleFree(const base::PlannerTerminationCondition &ptc)
208 {
209  unsigned int nodeCount = 0;
210  unsigned int sampleAttempts = 0;
211  Motion *motion = new Motion(si_);
212 
213  // Sample numSamples_ number of nodes from the free configuration space
214  while (nodeCount < numSamples_ && !ptc)
215  {
216  sampler_->sampleUniform(motion->getState());
217  sampleAttempts++;
218 
219  bool collision_free = si_->isValid(motion->getState());
220 
221  if (collision_free)
222  {
223  nodeCount++;
224  nn_->add(motion);
225  motion = new Motion(si_);
226  } // If collision free
227  } // While nodeCount < numSamples
228  si_->freeState(motion->getState());
229  delete motion;
230 
231  // 95% confidence limit for an upper bound for the true free space volume
232  freeSpaceVolume_ = boost::math::binomial_distribution<>::find_upper_bound_on_p(sampleAttempts, nodeCount, 0.05) * si_->getStateSpace()->getMeasure();
233 }

235 void ompl::geometric::FMT::assureGoalIsSampled(const ompl::base::GoalSampleableRegion *goal)
236 {
237  // Ensure that there is at least one node near each goal
238  while (const base::State *goalState = pis_.nextGoal())
239  {
240  Motion *gMotion = new Motion(si_);
241  si_->copyState(gMotion->getState(), goalState);
242 
243  std::vector<Motion*> nearGoal;
244  nn_->nearestR(gMotion, goal->getThreshold(), nearGoal);
245 
246  // If there is no node in the goal region, insert one
247  if (nearGoal.empty())
248  {
249  OMPL_DEBUG("No state inside goal region");
250  if (si_->getStateValidityChecker()->isValid(gMotion->getState()))
251  {
252  nn_->add(gMotion);
253  goalState_ = gMotion->getState();
254  }
255  else
256  {
257  si_->freeState(gMotion->getState());
258  delete gMotion;
259  }
260  }
261  else // There is already a sample in the goal region
262  {
263  goalState_ = nearGoal[0]->getState();
264  si_->freeState(gMotion->getState());
265  delete gMotion;
266  }
267  } // For each goal
268 }

270 ompl::base::PlannerStatus ompl::geometric::FMT::solve(const base::PlannerTerminationCondition &ptc)
271 {
272  if (lastGoalMotion_) {
273  OMPL_INFORM("solve() called before clear(); returning previous solution");
274  traceSolutionPathThroughTree(lastGoalMotion_);
275  OMPL_DEBUG("Final path cost: %f", lastGoalMotion_->getCost().value());
276  return base::PlannerStatus(true, false);
277  }
278  else if (Open_.size() > 0)
279  {
280  OMPL_INFORM("solve() called before clear(); no previous solution so starting afresh");
281  clear();
282  }
283 
284  checkValidity();
285  base::GoalSampleableRegion *goal = dynamic_cast<base::GoalSampleableRegion*>(pdef_->getGoal().get());
286  Motion *initMotion = NULL;
287 
288  if (!goal)
289  {
290  OMPL_ERROR("%s: Unknown type of goal", getName().c_str());
291  return base::PlannerStatus::UNRECOGNIZED_GOAL_TYPE;
292  }
293 
294  // Add start states to V (nn_) and Open
295  while (const base::State *st = pis_.nextStart())
296  {
297  initMotion = new Motion(si_);
298  si_->copyState(initMotion->getState(), st);
299  Open_.insert(initMotion);
300  initMotion->setSetType(Motion::SET_OPEN);
301  initMotion->setCost(opt_->initialCost(initMotion->getState()));
302  nn_->add(initMotion); // V <-- {x_init}
303  }
304 
305  if (!initMotion)
306  {
307  OMPL_ERROR("Start state undefined");
308  return base::PlannerStatus::INVALID_START;
309  }
310 
311  // Sample N free states in the configuration space
312  if (!sampler_)
313  sampler_ = si_->allocStateSampler();
314  sampleFree(ptc);
315  assureGoalIsSampled(goal);
316  OMPL_INFORM("%s: Starting planning with %u states already in datastructure", getName().c_str(), nn_->size());
317 
318  // Calculate the nearest neighbor search radius
320  if (nearestK_)
321  {
322  NNk_ = std::ceil(std::pow(2.0 * radiusMultiplier_, (double)si_->getStateDimension()) *
323  (boost::math::constants::e<double>() / (double)si_->getStateDimension()) *
324  log((double)nn_->size()));
325  OMPL_DEBUG("Using nearest-neighbors k of %d", NNk_);
326  }
327  else
328  {
329  NNr_ = calculateRadius(si_->getStateDimension(), nn_->size());
330  OMPL_DEBUG("Using radius of %f", NNr_);
331  }
332 
333  // Execute the planner, and return early if the planner returns a failure
334  bool plannerSuccess = false;
335  bool successfulExpansion = false;
336  Motion *z = initMotion; // z <-- xinit
337  saveNeighborhood(z);
338 
339  while (!ptc && !(plannerSuccess = goal->isSatisfied(z->getState())))
340  {
341  successfulExpansion = expandTreeFromNode(&z);
342  if (!successfulExpansion)
343  return base::PlannerStatus(false, false);
344  } // While not at goal
345 
346  if (plannerSuccess)
347  {
348  // Return the path to z, since by definition of planner success, z is in the goal region
349  lastGoalMotion_ = z;
350  traceSolutionPathThroughTree(lastGoalMotion_);
351 
352  OMPL_DEBUG("Final path cost: %f", lastGoalMotion_->getCost().value());
353 
354  return base::PlannerStatus(true, false);
355  } // if plannerSuccess
356  else
357  {
358  // Planner terminated without accomplishing goal
359  return base::PlannerStatus(false, false);
360  }
361 }

363 void ompl::geometric::FMT::traceSolutionPathThroughTree(Motion *goalMotion)
364 {
365  std::vector<Motion*> mpath;
366  Motion *solution = goalMotion;
367 
368  // Construct the solution path
369  while (solution != NULL)
370  {
371  mpath.push_back(solution);
372  solution = solution->getParent();
373  }
374 
375  // Set the solution path
376  PathGeometric *path = new PathGeometric(si_);
377  int mPathSize = mpath.size();
378  for (int i = mPathSize - 1 ; i >= 0 ; --i)
379  path->append(mpath[i]->getState());
380  pdef_->addSolutionPath(base::PathPtr(path), false, -1.0, getName());
381 }

383 bool ompl::geometric::FMT::expandTreeFromNode(Motion **z)
384 {
385  // Find all nodes that are near z, and also in set Unvisited
386 
387  std::vector<Motion*> xNear;
388  const std::vector<Motion*> &zNeighborhood = neighborhoods_[*z];
389  const unsigned int zNeighborhoodSize = zNeighborhood.size();
390  xNear.reserve(zNeighborhoodSize);
391 
392  for (unsigned int i = 0; i < zNeighborhoodSize; ++i)
393  {
394  Motion *x = zNeighborhood[i];
395  if (x->getSetType() == Motion::SET_UNVISITED)
396  {
397  saveNeighborhood(x);
398  if (nearestK_)
399  {
400  // Only include neighbors that are mutually k-nearest
401  // Relies on NN datastructure returning k-nearest in sorted order
402  const base::Cost connCost = opt_->motionCost((*z)->getState(), x->getState());
403  const base::Cost worstCost = opt_->motionCost(neighborhoods_[x].back()->getState(), x->getState());
404 
405  if (opt_->isCostBetterThan(worstCost, connCost))
406  continue;
407  else
408  xNear.push_back(zNeighborhood[i]);
409  }
410  else
411  xNear.push_back(x);
412  }
413  }
414 
415  // For each node near z and in set Unvisited, attempt to connect it to set Open
416  std::vector<Motion*> yNear;
417  std::vector<Motion*> Open_new;
418  const unsigned int xNearSize = xNear.size();
419  for (unsigned int i = 0 ; i < xNearSize; ++i)
420  {
421  Motion *x = xNear[i];
422 
423  // Find all nodes that are near x and in set Open
424  const std::vector<Motion*> &xNeighborhood = neighborhoods_[x];
425 
426  const unsigned int xNeighborhoodSize = xNeighborhood.size();
427  yNear.reserve(xNeighborhoodSize);
428  for (unsigned int j = 0; j < xNeighborhoodSize; ++j)
429  {
430  if (xNeighborhood[j]->getSetType() == Motion::SET_OPEN)
431  yNear.push_back(xNeighborhood[j]);
432  }
433 
434  // Find the lowest cost-to-come connection from Open to x
435  Motion *yMin = NULL;
436  base::Cost cMin(std::numeric_limits<double>::infinity());
437  const unsigned int yNearSize = yNear.size();
438  for (unsigned int j = 0; j < yNearSize; ++j)
439  {
440  const base::State *yState = yNear[j]->getState();
441  const base::Cost dist = opt_->motionCost(yState, x->getState());
442  const base::Cost cNew = opt_->combineCosts(yNear[j]->getCost(), dist);
443 
444  if (opt_->isCostBetterThan(cNew, cMin))
445  {
446  yMin = yNear[j];
447  cMin = cNew;
448  }
449  }
450  yNear.clear();
451 
452  // If an optimal connection from Open to x was found
453  if (yMin != NULL)
454  {
455  bool collision_free = false;
456  if (cacheCC_)
457  {
458  if (!yMin->alreadyCC(x))
459  {
460  collision_free = si_->checkMotion(yMin->getState(), x->getState());
461  ++collisionChecks_;
462  // Due to FMT* design, it is only necessary to save unsuccesful
463  // connection attemps because of collision
464  if (!collision_free)
465  yMin->addCC(x);
466  }
467  }
468  else
469  {
470  ++collisionChecks_;
471  collision_free = si_->checkMotion(yMin->getState(), x->getState());
472  }
473 
474  if (collision_free)
475  {
476  // Add edge from yMin to x
477  x->setParent(yMin);
478  x->setCost(cMin);
479  x->setHeuristicCost(opt_->motionCostHeuristic(x->getState(), goalState_));
480  // Add x to Open
481  Open_new.push_back(x);
482  // Remove x from Unvisited
483  x->setSetType(Motion::SET_CLOSED);
484  }
485  } // An optimal connection from Open to x was found
486  } // For each node near z and in set Unvisited, try to connect it to set Open
487 
488  // Update Open
489  Open_.pop();
490  (*z)->setSetType(Motion::SET_CLOSED);
491 
492  // Add the nodes in H_new to H
493  unsigned int openNewSize = Open_new.size();
494  for (unsigned int i = 0; i < openNewSize; i++)
495  {
496  Open_.insert(Open_new[i]);
497  Open_new[i]->setSetType(Motion::SET_OPEN);
498  }
499  Open_new.clear();
500 
501  if (Open_.empty())
502  {
503  OMPL_INFORM("Open is empty before path was found --> no feasible path exists");
504  return false;
505  }
506 
507  // Take the top of Open as the new z
508  *z = Open_.top()->data;
509 
510  return true;
511 }
