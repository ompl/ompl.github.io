172  void BITstar::setup()
173  {
174  //Call the base class setup:
175  Planner::setup();
176 
177  //Do some sanity checks
178  //Make sure we have a problem definition
179  if(Planner::pdef_ == false)
180  {
181  OMPL_ERROR("%s::setup() was called without a problem definition.", Planner::getName().c_str());
182  Planner::setup_ = false;
183  return;
184  }
185 
186  //Make sure the problem only has one start state.
187  if (Planner::pdef_->getStartStateCount() != 1u)
188  {
189  OMPL_ERROR("%s::setup() was called with %u start states, instead of exactly 1.", Planner::getName().c_str(), Planner::pdef_->getStartStateCount());
190  Planner::setup_ = false;
191  return;
192  }
193 
194  //Make sure we have an optimization objective
195  if (Planner::pdef_->hasOptimizationObjective() == false)
196  {
197  OMPL_INFORM("%s: No optimization objective specified. Defaulting to optimizing path length.", Planner::getName().c_str());
198  Planner::pdef_->setOptimizationObjective( boost::make_shared<base::PathLengthOptimizationObjective> (Planner::si_) );
199  }
200 
201  //Store the optimization objective for future ease of use
202  opt_ = Planner::pdef_->getOptimizationObjective();
203 
204  //Configure the nearest-neighbour constructs.
205  //Only allocate if they are empty (as they can be set to a specific version by a call to setNearestNeighbors)
206  if (static_cast<bool>(freeStateNN_) == false)
207  {
208  freeStateNN_.reset( ompl::tools::SelfConfig::getDefaultNearestNeighbors<VertexPtr>(Planner::si_->getStateSpace()) );
209  }
210  //No else, already allocated (by a call to setNearestNeighbors())
211 
212  if (static_cast<bool>(vertexNN_) == false)
213  {
214  vertexNN_.reset( ompl::tools::SelfConfig::getDefaultNearestNeighbors<VertexPtr>(Planner::si_->getStateSpace()) );
215  }
216  //No else, already allocated (by a call to setNearestNeighbors())
217 
218  //Configure:
219  freeStateNN_->setDistanceFunction(boost::bind(&BITstar::nnDistance, this, _1, _2));
220  vertexNN_->setDistanceFunction(boost::bind(&BITstar::nnDistance, this, _1, _2));
221 
222  //Create the start as a vertex:
223  startVertex_ = boost::make_shared<Vertex>(Planner::si_, opt_, true);
224 
225  //Copy the value of the start
226  Planner::si_->copyState(startVertex_->state(), pdef_->getStartState(0u));
227 
228  //Create the goal as a vertex:
229  //Create a vertex
230  goalVertex_ = boost::make_shared<Vertex>(Planner::si_, opt_);
231 
232  //Copy the value of the goal
233  Planner::si_->copyState(goalVertex_->state(), Planner::pdef_->getGoal()->as<ompl::base::GoalState>()->getState());
234 
235  //Configure the queue
236  //boost::make_shared can only take 9 arguments, so be careful:
237  intQueue_ = boost::make_shared<IntegratedQueue> (startVertex_, goalVertex_, boost::bind(&BITstar::nearestSamples, this, _1, _2), boost::bind(&BITstar::nearestVertices, this, _1, _2), boost::bind(&BITstar::lowerBoundHeuristicVertex, this, _1), boost::bind(&BITstar::currentHeuristicVertex, this, _1), boost::bind(&BITstar::lowerBoundHeuristicEdge, this, _1), boost::bind(&BITstar::currentHeuristicEdge, this, _1), boost::bind(&BITstar::currentHeuristicEdgeTarget, this, _1));
238  intQueue_->setUseFailureTracking(useFailureTracking_);
239  intQueue_->setDelayedRewiring(delayRewiring_);
240 
241  //Allocate a sampler:
242  sampler_ = opt_->allocInformedStateSampler(Planner::pdef_, std::numeric_limits<unsigned int>::max());
243 
244  //Set the best-cost and pruned-cost to the proper opt_-based values:
245  bestCost_ = opt_->infiniteCost();
246  prunedCost_ = opt_->infiniteCost();
247 
248  //Set the minimum cost as the cost-to-come of the goal:
249  minCost_ = this->costToComeHeuristic(goalVertex_);
250 
251  //Set the cost sampled to the maximum. This forces us to first check the basic start-goal graph.
252  costSampled_ = bestCost_;
253 
254  //Finally, as they depend on some of the above, insert the start and goal into the proper sets
255  //Add the goal to the set of samples.:
256  this->addSample(goalVertex_);
257 
258  //Store the start into the tree and the queue:
259  this->addVertex(startVertex_, false);
260 
261  //Finally initialize the nearestNeighbour terms:
262  this->initializeNearestTerms();
263 
264  //Debug: Output an estimate of the state measure:
265 // this->estimateMeasures();
266  }

350  ompl::base::PlannerStatus BITstar::solve(const ompl::base::PlannerTerminationCondition& ptc)
351  {
352  Planner::checkValidity();
353  OMPL_INFORM("%s: Searching for a solution to the given planning problem.", Planner::getName().c_str());
354  this->statusMessage(ompl::msg::LOG_DEBUG, "Start solve");
355 
356  //Reset the manual stop to the iteration loop:
357  stopLoop_ = false;
358 
359  //Run the outerloop until we're stopped, a suitable cost is found, or until we find the minimum possible cost within tolerance:
360  while (opt_->isSatisfied(bestCost_) == false && ptc == false && this->isCostBetterThan(minCost_, bestCost_) == true && stopLoop_ == false)
361  {
362  this->iterate();
363  }
364 
365  if (hasSolution_ == true)
366  {
367  this->endSuccessMessage();
368 
369  this->publishSolution();
370  }
371  else
372  {
373  this->endFailureMessage();
374  }
375 
376  this->statusMessage(ompl::msg::LOG_DEBUG, "End solve");
377 
378  //PlannerStatus(addedSolution, approximate)
379  return ompl::base::PlannerStatus(hasSolution_, approximateSoln_);
380  }

384  void BITstar::getPlannerData(ompl::base::PlannerData& data) const
385  {
386  //Get the base planner class data:
387  Planner::getPlannerData(data);
388 
389  //Add samples
390  if (freeStateNN_)
391  {
392  //Variables:
393  //The list of unused samples:
394  std::vector<VertexPtr> samples;
395 
396  //Get the list of samples
397  freeStateNN_->list(samples);
398 
399  //Iterate through it turning each into a disconnected vertex
400  for (std::vector<VertexPtr>::const_iterator sIter = samples.begin(); sIter != samples.end(); ++sIter)
401  {
402  //No, add as a regular vertex:
403  data.addVertex(ompl::base::PlannerDataVertex((*sIter)->stateConst()));
404  }
405  }
406  //No else.
407 
408  //Add vertices
409  if (vertexNN_)
410  {
411  //Variables:
412  //The list of vertices in the graph:
413  std::vector<VertexPtr> vertices;
414 
415  //Get the list of vertices
416  vertexNN_->list(vertices);
417 
418  //Iterate through it turning each into a vertex with an edge:
419  for (std::vector<VertexPtr>::const_iterator vIter = vertices.begin(); vIter != vertices.end(); ++vIter)
420  {
421  //Is the vertex the start?
422  if ((*vIter)->isRoot() == true)
423  {
424  //Yes, add as a start vertex:
425  data.addStartVertex(ompl::base::PlannerDataVertex((*vIter)->stateConst()));
426  }
427  else
428  {
429  //No, add as a regular vertex:
430  data.addVertex(ompl::base::PlannerDataVertex((*vIter)->stateConst()));
431 
432  //And as an incoming edge
433  data.addEdge(ompl::base::PlannerDataVertex((*vIter)->getParentConst()->stateConst()), ompl::base::PlannerDataVertex((*vIter)->stateConst()));
434  }
435  }
436  }
437  //No else.
438 
439  //Did we find a solution?
440  if (hasSolution_ == true)
441  {
442  data.markGoalState(goalVertex_->stateConst());
443  }
444  }

541  void BITstar::estimateMeasures()
542  {
543  OMPL_INFORM("%s: Estimating the measure of the planning domain. This is a debugging function that does not have any effect on the planner.", Planner::getName().c_str());
544  //Variables:
545  //The total number of samples:
546  unsigned int numTotalSamples;
547  //The resulting samples in free:
548  unsigned int numFreeSamples;
549  //The resulting samples in obs:
550  unsigned int numObsSamples;
551  //The sample fraction of free:
552  double fractionFree;
553  //The sample fraction of obs:
554  double fractionObs;
555  //The total measure of the space:
556  double totalMeasure;
557  //The resulting estimate of the free measure
558  double freeMeasure;
559  //The resulting estimate of the obs measure
560  double obsMeasure;
561 
562  //Set the total number of samples
563  numTotalSamples = 100000u;
564  numFreeSamples = 0u;
565  numObsSamples = 0u;
566 
567  //Draw samples, classifying each one
568  for (unsigned int i = 0u; i < numTotalSamples; ++i)
569  {
570  //Allocate a state
571  ompl::base::State* aState = Planner::si_->allocState();
572 
573  //Sample:
574  sampler_->sampleUniform(aState, bestCost_);
575 
576  //Check if collision free
577  if (Planner::si_->isValid(aState) == true)
578  {
579  ++numFreeSamples;
580  }
581  else
582  {
583  ++numObsSamples;
584  }
585  }
586 
587  //Calculate the fractions:
588  fractionFree = static_cast<double>(numFreeSamples)/static_cast<double>(numTotalSamples);
589 
590  fractionObs = static_cast<double>(numObsSamples)/static_cast<double>(numTotalSamples);
591 
592  //Get the total measure of the space
593  totalMeasure = Planner::si_->getSpaceMeasure();
594 
595  //Calculate the measure of the free space
596  freeMeasure = fractionFree*totalMeasure;
597 
598  //Calculate the measure of the obs space
599  obsMeasure = fractionObs*totalMeasure;
600 
601  //Announce
602  OMPL_INFORM("%s: %u samples (%u free, %u in collision) from a space with measure %.4f estimates %.2f%% free and %.2f%% in collision (measures of %.4f and %.4f, respectively).", Planner::getName().c_str(), numTotalSamples, numFreeSamples, numObsSamples, totalMeasure, 100.0*fractionFree, 100.0*fractionObs, freeMeasure, obsMeasure);
603  }

614  if(useStrictQueueOrdering_ == true)
615  {
616  //The queues will be resorted if the graph has been rewired.
617  this->resort();
618  }
619 
620  //If the edge queue is empty, that must mean we're either starting from scratch, or just finished a batch. Either way, make a batch of samples and fill the queue for the first time:
621  if (intQueue_->isEmpty() == true)
622  {
623  this->newBatch();
624  }
625  else
626  {
627  //If the edge queue is not empty, then there is work to do!
628 
629  //Variables:
630  //The current edge:
631  VertexPtrPair bestEdge;
632 
633  //Pop the minimum edge
634  intQueue_->popFrontEdge(&bestEdge);
635 
636  //In the best case, can this edge improve our solution given the current graph?
637  //g_t(v) + c_hat(v,x) + h_hat(x) < g_t(x_g)
638  if (this->isCostBetterThan( this->combineCosts(bestEdge.first->getCost(), this->edgeCostHeuristic(bestEdge), this->costToGoHeuristic(bestEdge.second)), bestCost_ ) == true)
639  {
640  //Variables:
641  //The true cost of the edge:
642  ompl::base::Cost trueEdgeCost;
643 
644  //Get the true cost of the edge
645  trueEdgeCost = this->trueEdgeCost(bestEdge);
646 
647  //Can this actual edge ever improve our solution?
648  //g_hat(v) + c(v,x) + h_hat(x) < g_t(x_g)
649  if (this->isCostBetterThan( this->combineCosts(this->costToComeHeuristic(bestEdge.first), trueEdgeCost, this->costToGoHeuristic(bestEdge.second)), bestCost_ ) == true)
650  {
651  //Does this edge have a collision?
652  if (this->checkEdge(bestEdge) == true)
653  {
654  //Does the current edge improve our graph?
655  //g_t(v) + c(v,x) < g_t(x)
656  if (this->isCostBetterThan( opt_->combineCosts(bestEdge.first->getCost(), trueEdgeCost), bestEdge.second->getCost() ) == true)
657  {
658  //YAAAAH. Add the edge! Allowing for the sample to be removed from free if it is not currently connected and otherwise propagate cost updates to descendants.
659  //addEdge will update the queue and handle the extra work that occurs if this edge improves the solution.
660  this->addEdge(bestEdge, trueEdgeCost, true, true);
661 
662  //Prune the edge queue of any unnecessary incoming edges
663  intQueue_->pruneEdgesTo(bestEdge.second);
664 
665  //We will only prune the whole graph/samples on a new batch.
666  }
667  //No else, this edge may be useful at some later date.
668  }
669  else if (useFailureTracking_ == true)
670  {
671  //If the edge failed, and we're tracking failures, record.
672  //This edge has a collision and can never be helpful. Poor edge. Add the target to the list of failed children for the source:
673  bestEdge.first->markAsFailedChild(bestEdge.second);
674  }
675  //No else, we failed and we're not tracking those
676  }
677  else if (useFailureTracking_ == true)
678  {
679  //If the edge failed, and we're tracking failures, record.
680  //This edge either has a very high edge cost and can never be helpful. Poor edge. Add the target to the list of failed children for the source
681  bestEdge.first->markAsFailedChild(bestEdge.second);
682  }
683  //No else, we failed and we're not tracking those
684  }
685  else if (intQueue_->isSorted() == false)
686  {
687  //The edge cannot improve our solution, but the queue is imperfectly sorted, so we must resort before we give up.
688  this->resort();
689  }
690  else
691  {
692  this->statusMessage(ompl::msg::LOG_DEBUG, "Clearing queue!");
693  //Else, I cannot improve the current solution, and as the queue is perfectly sorted and I am the best edge, no one can improve the current solution . Give up on the batch:
694  intQueue_->finish();
695  }
696  } //Integrated queue not empty.
697  }
