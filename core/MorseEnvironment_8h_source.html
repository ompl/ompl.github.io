118  // Replace infinite bounds with very large bounds, so, e.g., sampling can still work
119  for (unsigned int i = 0; i < positionBounds_.size(); i++)
120  {
121  if (positionBounds_[i] == std::numeric_limits<double>::infinity())
122  positionBounds_[i] = std::numeric_limits<double>::max() / 2;
123  else if (positionBounds_[i] == -std::numeric_limits<double>::infinity())
124  positionBounds_[i] = -std::numeric_limits<double>::max() / 2;
125  }
126  for (unsigned int i = 0; i < linvelBounds_.size(); i++)
127  {
128  if (linvelBounds_[i] == std::numeric_limits<double>::infinity())
129  linvelBounds_[i] = std::numeric_limits<double>::max() / 2;
130  else if (linvelBounds_[i] == -std::numeric_limits<double>::infinity())
131  linvelBounds_[i] = -std::numeric_limits<double>::max() / 2;
132  }
133  for (unsigned int i = 0; i < angvelBounds_.size(); i++)
134  {
135  if (angvelBounds_[i] == std::numeric_limits<double>::infinity())
136  angvelBounds_[i] = std::numeric_limits<double>::max() / 2;
137  else if (angvelBounds_[i] == -std::numeric_limits<double>::infinity())
138  angvelBounds_[i] = -std::numeric_limits<double>::max() / 2;
