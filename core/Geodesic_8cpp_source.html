59double BundleSpaceMetricGeodesic::distanceBase(const Configuration *xStart, const Configuration *xDest)
60{
61 if (bundleSpaceGraph_->getBaseDimension() > 0)
62 {
63 bundleSpaceGraph_->getProjection()->project(xStart->state, xBaseStartTmp_);
64 bundleSpaceGraph_->getProjection()->project(xDest->state, xBaseDestTmp_);
65 double d = bundleSpaceGraph_->getBase()->distance(xBaseStartTmp_, xBaseDestTmp_);
66 return d;
67 }
68 else
69 {
70 return 0.0;
71 }
72}
