440  if (heuristics_)
441  return ( opt_->combineCosts(p1->getCost(), p1->getHeuristicCost()).value() < opt_->combineCosts(p2->getCost(), p2->getHeuristicCost()).value() );

443  return (p1->getCost().value() < p2->getCost().value());
444  }
