79 ompl::ProlateHyperspheroid::ProlateHyperspheroid(unsigned int n, const double focus1[], const double focus2[])
80  : dataPtr_ (boost::make_shared<PhsData>())
81 {
82  //Initialize the data:
83  dataPtr_->dim_ = n;
84  dataPtr_->transverseDiameter_ = 0.0; // Initialize to something.
85  dataPtr_->isTransformUpToDate_ = false;
86 
87  // Copy the arrays into their Eigen containers via the Eigen::Map "view"
88  dataPtr_->xFocus1_ = Eigen::Map<const Eigen::VectorXd>(focus1, dataPtr_->dim_);
89  dataPtr_->xFocus2_ = Eigen::Map<const Eigen::VectorXd>(focus2, dataPtr_->dim_);
90 
91  // Calculate the minimum transverse diameter
92  dataPtr_->minTransverseDiameter_ = (dataPtr_->xFocus1_ - dataPtr_->xFocus2_).norm();
93 
94  // Calculate the centre:
95  dataPtr_->xCentre_ = 0.5 * (dataPtr_->xFocus1_ + dataPtr_->xFocus2_);
96 
97  // Calculate the rotation
98  updateRotation();
99 }

101 void ompl::ProlateHyperspheroid::setTransverseDiameter(double transverseDiameter)
102 {
103  if (transverseDiameter < dataPtr_->minTransverseDiameter_)
104  {
105  OMPL_ERROR("%g < %g", transverseDiameter, dataPtr_->minTransverseDiameter_);
106  throw Exception("Transverse diameter cannot be less than the distance between the foci.");
107  }
108 
109  // Store and update if changed
110  if (dataPtr_->transverseDiameter_ != transverseDiameter)
111  {
112  // Mark as out of date
113  dataPtr_->isTransformUpToDate_ = false;
114 
115  // Store
116  dataPtr_->transverseDiameter_ = transverseDiameter;
117 
118  // Update the transform
119  updateTransformation();
120  }
121  // No else, the diameter didn't change
122 }

181 double ompl::ProlateHyperspheroid::calcPhsMeasure(unsigned int N, double minTransverseDiameter, double transverseDiameter)
182 {
183  if (transverseDiameter < minTransverseDiameter)
184  {
185  throw Exception("Transverse diameter cannot be less than the minimum transverse diameter.");
186  }
187  // Variable
188  // The conjugate diameter:
189  double conjugateDiameter;
190  // The Lebesgue measure return value
191  double lmeas;
192 
193  // Calculate the conjugate diameter:
194  conjugateDiameter = std::sqrt(transverseDiameter * transverseDiameter - minTransverseDiameter * minTransverseDiameter);
195 
196  // Calculate as a product series of the radii, noting that one is the transverse diameter/2.0, and the other N-1 are the conjugate diameter/2.0
197  lmeas = transverseDiameter/2.0;
198  for (unsigned int i = 1u; i < N; ++i)
199  {
200  lmeas = lmeas * conjugateDiameter/2.0;
201  }
202 
203  // Then multiplied by the volume of the unit n-ball.
204  lmeas = lmeas * unitNBallMeasure(N);
205 
206  // Return:
207  return lmeas;
208 }

215 void ompl::ProlateHyperspheroid::updateRotation(void)
216 {
217  // Mark the transform as out of date
218  dataPtr_->isTransformUpToDate_ = false;
219 
220  // If the minTransverseDiameter_ is too close to 0, we treat this as a circle.
221  double circleTol = 1E-9;
222  if (dataPtr_->minTransverseDiameter_ < circleTol)
223  {
224  dataPtr_->rotationWorldFromEllipse_.setIdentity(dataPtr_->dim_, dataPtr_->dim_);
225  }
226  else
227  {
228  // Variables
229  // The transverse axis of the PHS expressed in the world frame.
230  Eigen::VectorXd transverseAxis(dataPtr_->dim_);
231  // The matrix representation of the Wahba problem
232  Eigen::MatrixXd wahbaProb(dataPtr_->dim_, dataPtr_->dim_);
233  // The middle diagonal matrix in the SVD solution to the Wahba problem
234  Eigen::VectorXd middleM(dataPtr_->dim_);
235 
236  // Calculate the major axis, storing as the first eigenvector
237  transverseAxis = (dataPtr_->xFocus2_ - dataPtr_->xFocus1_ )/dataPtr_->minTransverseDiameter_;
238 
239  // Calculate the rotation that will allow us to generate the remaining eigenvectors
240  // Formulate as a Wahba problem, first forming the matrix a_j*a_i' where a_j is the transverse axis if the ellipse in the world frame, and a_i is the first basis vector of the world frame (i.e., [1 0 .... 0])
241  wahbaProb = transverseAxis * Eigen::MatrixXd::Identity(dataPtr_->dim_, dataPtr_->dim_).col(0).transpose();
242 
243  // Then run it through the SVD solver
244  Eigen::JacobiSVD<Eigen::MatrixXd, Eigen::NoQRPreconditioner> svd(wahbaProb, Eigen::ComputeFullV | Eigen::ComputeFullU);
245 
246  // Then calculate the rotation matrix from the U and V components of SVD
247  // Calculate the middle diagonal matrix
248  middleM = Eigen::VectorXd::Ones(dataPtr_->dim_);
249  // Make the last value equal to det(U)*det(V) (zero-based indexing remember)
250  middleM(dataPtr_->dim_ - 1) = svd.matrixU().determinant() * svd.matrixV().determinant();
251 
252  // Calculate the rotation
253  dataPtr_->rotationWorldFromEllipse_ = svd.matrixU() * middleM.asDiagonal() * svd.matrixV().transpose();
254  }
255 }

257 void ompl::ProlateHyperspheroid::updateTransformation(void)
258 {
259  // Variables
260  // The radii of the ellipse
261  Eigen::VectorXd diagAsVector(dataPtr_->dim_);
262  // The conjugate diameters:
263  double conjugateDiamater;
264 
265  // Calculate the conjugate radius
266  conjugateDiamater = std::sqrt(dataPtr_->transverseDiameter_*dataPtr_->transverseDiameter_ - dataPtr_->minTransverseDiameter_*dataPtr_->minTransverseDiameter_);
267 
268  // Store into the diagonal matrix
269  // All the elements but one are the conjugate radius
270  diagAsVector.fill(conjugateDiamater/2.0);
271 
272  // The first element in diagonal is the transverse radius
273  diagAsVector(0) = 0.5 * dataPtr_->transverseDiameter_;
274 
275  // Calculate the transformation matrix
276  dataPtr_->transformationWorldFromEllipse_ = dataPtr_->rotationWorldFromEllipse_ * diagAsVector.asDiagonal();
277 
278  // Calculate the measure:
279  dataPtr_->phsMeasure_ = calcPhsMeasure(dataPtr_->dim_, dataPtr_->minTransverseDiameter_, dataPtr_->transverseDiameter_);
280 
281  // Mark as up to date
282  dataPtr_->isTransformUpToDate_ = true;
283 }
