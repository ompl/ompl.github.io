56 bool isStateValid(double radiusSquared, const ob::State *state)
57 {
58  const ob::SE2StateSpace::StateType *s = state->as<ob::SE2StateSpace::StateType>();
59  double x=s->getX(), y=s->getY();
60  x = std::abs(x - std::floor(x));
61  y = std::abs(y - std::floor(y));
62  x = std::min(x, 1. - x);
63  y = std::min(y, 1. - y);
64  return x*x + y*y > radiusSquared;
65 }

67 int main(int argc, char **argv)
68 {
69  int distance, gridLimit, runCount;
70  double obstacleRadius, turningRadius, runtimeLimit;
71 
72  ob::StateSpacePtr space(new ob::SE2StateSpace());
73 
74  po::options_description desc("Options");
75 
76  desc.add_options()
77  ("help", "show help message")
78  ("dubins", "use Dubins state space")
79  ("dubinssym", "use symmetrized Dubins state space")
80  ("reedsshepp", "use Reeds-Shepp state space")
81  ("distance", po::value<int>(&distance)->default_value(3), "integer grid distance between start and goal")
82  ("obstacle-radius", po::value<double>(&obstacleRadius)->default_value(.25), "radius of obstacles")
83  ("turning-radius", po::value<double>(&turningRadius)->default_value(.5), "turning radius of robot (ignored for default point robot)")
84  ("grid-limit", po::value<int>(&gridLimit)->default_value(10), "size of the grid")
85  ("runtime-limit", po::value<double>(&runtimeLimit)->default_value(2), "time limit for every test")
86  ("run-count", po::value<int>(&runCount)->default_value(100), "number of times to run each planner")
87  ;
88 
89  po::variables_map vm;
90  po::store(po::parse_command_line(argc, argv, desc), vm);
91  po::notify(vm);
92 
93  if (vm.count("help"))
94  {
95  std::cout << desc << "\n";
96  return 1;
97  }
98 
99  if (vm.count("dubins"))
100  space = ob::StateSpacePtr(new ob::DubinsStateSpace(turningRadius));
101  if (vm.count("dubinssym"))
102  space = ob::StateSpacePtr(new ob::DubinsStateSpace(turningRadius, true));
103  if (vm.count("reedsshepp"))
104  space = ob::StateSpacePtr(new ob::ReedsSheppStateSpace(turningRadius));
105 
106  // set the bounds for the R^2 part of SE(2)
107  ob::RealVectorBounds bounds(2);
108  bounds.setLow(-.5 * gridLimit);
109  bounds.setHigh(.5 * gridLimit);
110  space->as<ob::SE2StateSpace>()->setBounds(bounds);
111 
112  // define a simple setup class
113  og::SimpleSetup ss(space);
114 
115  // set state validity checking for this space
116  ss.setStateValidityChecker(std::bind(&isStateValid, obstacleRadius*obstacleRadius, std::placeholders::_1));
117 
118  // define start & goal states
119  ob::ScopedState<ob::SE2StateSpace> start(space), goal(space);
120  start->setXY(0., 0.5);
121  start->setYaw(0.);
122  goal->setXY(0., (double)distance + .5);
123  goal->setYaw(0);
124  ss.setStartAndGoalStates(start, goal);
125 
126  // setting collision checking resolution to 0.05 (absolute)
127  ss.getSpaceInformation()->setStateValidityCheckingResolution(0.05 / gridLimit);
128  ss.getProblemDefinition()->setOptimizationObjective(getPathLengthObjective(ss.getSpaceInformation()));
129 
130  // by default, use the Benchmark class
131  double memoryLimit = 4096;
132  ot::Benchmark::Request request(runtimeLimit, memoryLimit, runCount);
133  ot::Benchmark b(ss, "CircleGrid");
134 
135  b.addPlanner(ob::PlannerPtr(new og::RRTstar(ss.getSpaceInformation())));
136  b.addPlanner(ob::PlannerPtr(new og::CForest(ss.getSpaceInformation())));
137  b.benchmark(request);
138  b.saveResultsToFile("circleGrid.log");
139 
140  exit(0);
141 }
