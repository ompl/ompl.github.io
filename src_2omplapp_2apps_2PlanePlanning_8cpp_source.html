93  private:
94  void liftState(ompl::base::State *state)
95  {
96  auto s0 = state->as<ompl::base::CompoundState>()->as<ompl::base::SE3StateSpace::StateType>(0);
97  const auto s1 = state->as<ompl::base::CompoundState>()->as<ompl::base::VanaStateSpace::StateType>(1);
98  s0->setXYZ((*s1)[0], (*s1)[1], (*s1)[2]);
99  Eigen::Quaterniond q = Eigen::AngleAxisd(s1->pitch(), Eigen::Vector3d::UnitY())
100  * Eigen::AngleAxisd(s1->yaw(), Eigen::Vector3d::UnitZ());
101  auto& qs = s0->rotation();
102  qs.w = q.w();
103  qs.x = q.x();
104  qs.y = q.y();
105  qs.z = q.z();
106  }
107  };

127 ompl::base::ScopedState<>
128 ompl::app::PlanePlanning::getFullStateFromGeometricComponent(const base::ScopedState<> &state) const
129 {
130  base::ScopedState<> s(getStateSpace());
131  // Eigen quaternion is ordered (w,x,y,z), while OMPL's quaternion is ordered
132  // x,y,z,w
133  auto euler = Eigen::Quaterniond(state[6], state[3], state[4], state[5]).toRotationMatrix().eulerAngles(0, 1, 2);
134  s[0] = s[7] = state[0];
135  s[1] = s[8] = state[1];
136  s[2] = s[9] = state[2];
137  s[3] = state[3];
138  s[4] = state[4];
139  s[5] = state[5];
140  s[6] = state[6];
141  s[10] = euler[1]; // pitch
142  s[11] = euler[2]; // yaw
143  return s;
144 }
