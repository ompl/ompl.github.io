68 # Create a narrow passage between y=[-3,3]. Only a 6x6x6 cube will be valid, centered at origin
69 def isStateValid(state):
70  if (state.getY() >= -3 and state.getY() <= 3):
71  if (state.getX() >= -3 and state.getX() <= 3 and state.getZ() >= -3 and state.getZ() <= 3):

74  return False

103  print("Strongly connected components: " + str(len(hist)))
104

106  comps, hist = gt.label_components(graph)
107  print("Weakly connected components: " + str(len(hist)))

109  # Plotting the graph
110  gt.remove_parallel_edges(graph) # Removing any superfluous edges
111 
112  edgeweights = graph.edge_properties["weight"]
113  colorprops = graph.new_vertex_property("string")
114  vertexsize = graph.new_vertex_property("double")

116  start = -1
117  goal = -1

119  for v in range(graph.num_vertices()):

121  # Color and size vertices by type: start, goal, other
122  if (pd.isStartVertex(v)):
123  start = v
124  colorprops[graph.vertex(v)] = "cyan"
125  vertexsize[graph.vertex(v)] = 10
126  elif (pd.isGoalVertex(v)):
127  goal = v
128  colorprops[graph.vertex(v)] = "green"
129  vertexsize[graph.vertex(v)] = 10

131  colorprops[graph.vertex(v)] = "yellow"

135  edgecolor = graph.new_edge_property("string")
136  edgesize = graph.new_edge_property("double")
137  for e in graph.edges():
138  edgecolor[e] = "black"
139  edgesize[e] = 0.5
140 
141  # using A* to find shortest path in planner data
142  if start != -1 and goal != -1:
143  dist, pred = gt.astar_search(graph, graph.vertex(start), edgeweights)
144 
145  # Color edges along shortest path red with size 3.0

163 def plan():
164  # construct the state space we are planning in
165  space = ob.SE3StateSpace()
166 
167  # set the bounds for R^3 portion of SE(3)
168  bounds = ob.RealVectorBounds(3)
169  bounds.setLow(-10)
170  bounds.setHigh(10)
171  space.setBounds(bounds)
172 
173  # define a simple setup class
174  ss = og.SimpleSetup(space)
175 
176  # create a start state
177  start = ob.State(space)
178  start().setX(-9)
179  start().setY(-9)
180  start().setZ(-9)
181  start().rotation().setIdentity()
182 
183  # create a goal state
184  goal = ob.State(space)
185  goal().setX(-9)
186  goal().setY(9)
187  goal().setZ(-9)
188  goal().rotation().setIdentity()
189 
190  ss.setStateValidityChecker(ob.StateValidityCheckerFn(isStateValid))
191 
192  # set the start and goal states
193  ss.setStartAndGoalStates(start, goal, 0.05)
194 
195  # Lets use PRM. It will have interesting PlannerData
196  planner = og.PRM(ss.getSpaceInformation())
197  ss.setPlanner(planner)
198  ss.setup()
199 
200  # attempt to solve the problem
201  solved = ss.solve(20.0)
202 
203  if solved:
204  # print the path to screen
205  print("Found solution:\n%s" % ss.getSolutionPath())
206 
207  # Extracting planner data from most recent solve attempt
208  pd = ob.PlannerData(ss.getSpaceInformation())
209  ss.getPlannerData(pd)
210 
211  # Computing weights of all edges based on state space distance
212  pd.computeEdgeWeights()
213 
214  if graphtool:
215  useGraphTool(pd, space)
