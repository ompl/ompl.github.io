57 double ompl::prolateHyperspheroidMeasure(unsigned int N, double dFoci, double dTransverse)
58 {
59  // Sanity check input
60  if (dTransverse < dFoci)
61  {
62  throw Exception("Transverse diameter cannot be less than the minimum transverse diameter.");
63  }
64 
65  // Variable
66  // The conjugate diameter:
67  double conjugateDiameter;
68  // The Lebesgue measure return value
69  double lmeas;
70 
71  // Calculate the conjugate diameter:
72  conjugateDiameter = std::sqrt(dTransverse * dTransverse - dFoci * dFoci);
73 
74  // Calculate the volume
75  // First multiply together the radii, noting that the first radius is the transverse diameter/2.0, and the other N-1 are the conjugate diameter/2.0
76  lmeas = dTransverse/2.0;
77  for (unsigned int i = 1u; i < N; ++i)
78  {
79  lmeas = lmeas * conjugateDiameter/2.0;
80  }
81 
82  // Then multiply by the volume of the unit n-ball.
83  lmeas = lmeas * unitNBallMeasure(N);
84 
85  // Finally return:
86  return lmeas;
87 }
