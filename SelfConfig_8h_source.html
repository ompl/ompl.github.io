94  static NearestNeighbors<_T>* getDefaultNearestNeighbors(const base::Planner *planner)
95  {
96  const base::StateSpacePtr &space = planner->getSpaceInformation()->getStateSpace();
97  const base::PlannerSpecs &specs = planner->getSpecs();
98  if (space->isMetricSpace())
99  {
100  if (specs.multithreaded)
101  return new NearestNeighborsGNAT<_T>();
102  else
103  return new NearestNeighborsGNATNoThreadSafety<_T>();
104  }
105  else
