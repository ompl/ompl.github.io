240  void solve(StateType &state, const Control *control, double duration) const override
241  {
242  ODESolver::ODEFunctor odefunc(ode_, control);
243 
244  if (error_.size() != state.size())
245  error_.assign(state.size(), 0.0);
246 
247  Solver solver;
248  solver.adjust_size(state);
249 
250  double time = 0.0;
251  while (time < duration + std::numeric_limits<float>::epsilon())
252  {
253  solver.do_step(odefunc, state, time, intStep_, error_);
254  time += intStep_;
255  }
256  }
257 
259  mutable ODESolver::StateType error_;
260  };

ompl::control::ODEAdaptiveSolver
Adaptive step size solver for ordinary differential equations of the type q&#39; = f(q, u), where q is the current state of the system and u is a control applied to the system. The maximum integration error is bounded in this approach. Solver is the numerical integration method used to solve the equations, and must implement the error stepper concept from boost::numeric::odeint. The default is a fifth order Runge-Kutta Cash-Karp method with a fourth order error bound.
