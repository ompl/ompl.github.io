95  virtual _T nearest(const _T &data) const
96  {
97  const std::size_t n = NearestNeighborsLinear<_T>::data_.size();
98  std::size_t pos = n;
99 
100  if (checks_ > 0 && n > 0)
101  {
102  double dmin = 0.0;
103  for (std::size_t j = 0 ; j < checks_ ; ++j)
104  {
105  std::size_t i = (j * checks_ + offset_) % n;
106 
107  double distance = NearestNeighbors<_T>::distFun_(NearestNeighborsLinear<_T>::data_[i], data);
108  if (pos == n || dmin > distance)
109  {
110  pos = i;
111  dmin = distance;
112  }
113  }
114  offset_ = (offset_ + 1) % checks_;
115  }
116  if (pos != n)
117  return NearestNeighborsLinear<_T>::data_[pos];
118 
119  throw Exception("No elements found in nearest neighbors data structure");
120  }
121
