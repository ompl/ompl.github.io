175  double distanceFunction(const Motion *a, const Motion *b) const
176  {
177  double sqr_dist = 0.0;
178  for (int ix = 0; ix < a->proj.size(); ++ix)
179  {
180  double sqr_val = (*b).proj[ix] - (*a).proj[ix];
181  sqr_val *= sqr_val;
182 
183  sqr_dist += sqr_val;
184  }
